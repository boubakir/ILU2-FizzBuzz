		assertEquals(FizzBuzz.fizzBuzz(3), "fizz");

		assertEquals(FizzBuzz.fizzBuzz(4), "4");

	}

	void test_cinq() {

	void test_multiples() {
		assertEquals(FizzBuzz.fizzBuzz(6), "fizz");

		assertEquals(FizzBuzz.fizzBuzz(10), "buzz");
